type
-- for indexing
CacheIndex : scalarset(5);

DirectoryIndex : scalarset(1);

Node: union { Cache, Directory };

VC_Type : 0..NUM_VCs -1;

-- Message Type
MessageType: enum { Ack,
                    Fwd };

-- Message
Message:
 record
  mtype : MessageType;
  src : Node
  src: Cache;
  dst: Cache;
  Acks_received: 0..5;
 end;

-- Machine States
CacheState:
 record
  state: enum { I,
                M,
                IM,
                S,
                MI };
  ack_rec: 0..5;
 end;

DirectoryState:
 record
  state: enum { I,
                M,
                IM };
  owner: Cache;,
  ack_rec: 0..5;
 end;
