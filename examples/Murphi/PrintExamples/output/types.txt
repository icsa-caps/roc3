type
-- for indexing
CacheInd : scalarset(5);

DirectoryInd : scalarset(1);


Node: union { Cache, Directory };

VC_Type : 0..NUM_VCs -1;

-- Message Type
MessageType: enum { Ack,
                    Fwd };

-- Message
Message:
 Record
  mtype : MessageType;
  src : Node
  src: Cache;
  dst: Cache;
  Acks_received: 0..5;
 end;

-- Machine States
CacheState:
 record
  state: enum { I,
                M,
                IM,
                S,
                MI };
  ack_rec: 0..5;
 end;

DirectoryState:
 record
  state: enum { I,
                M,
                IM };
  owner: Cache;,
  ack_rec: 0..5;
 end;
